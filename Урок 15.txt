void fill(int* mas, int n)
{
	for (int i = 0; i < n; i++)
	{
		mas[i] = rand() % 100 - 50;
	}
}

void sort(int* mass, int n)
{
	int temp = 0;
	for (int i = 1; i < n; ++i)
	{
		for (int r = 0; r < n - i; r++)
		{
			if (mass[r] < mass[r + 1])
			{
				swap(mass[r], mass[r + 1]);
			}
		}
	}
}

int max_array(int* mas, int n)
{
	int max = mas[0];
	for (int i = 1; i < n; i++)
	{
		if (max <= mas[i])
		{
			max = mas[i];
		}
	}
	return max;
}

int min_array(int* mas, int n)
{
	int min = mas[0];
	for (int i = 1; i < n; i++)
	{
		if (min >= mas[i])
		{
			min = mas[i];
		}
	}
	return min;
}

		}
		if (k < 0)
		{
			cout << "Error" << endl;
			continue;
		}
		int* mas = new int[size - 1];
		for (int i = 0; i < k; i++)
		{
			mas[i] = array[i];

			mas[i-1] = array[i];

		for (int i = 0; i < size; i++)

			{
				str.erase(str.begin() + i);
			}
		}
	}
	for (int i = 0; i < str.length(); i++)
	{
		if (str[i] == ' ')
		{
			if (str[i + 1] == ',')
			{
				str.erase(str.begin() + i);
			}
		}
	}
	for (int i = 0; i < str.length(); i++)

		if (str[i] == '.')
		{
			if (str[i + 1] == '.')
			{
				str.erase(str.begin() + i);
			}
		}
	}

		if (str[i] == ',')
		{
			if (str[i + 1] == ',')
			{
				str.erase(str.begin() + i);
			}
		}

#include <iostream>
#include <ctime>
#include <string>
#include <Windows.h>

using namespace std;

int main()
{
	setlocale(LC_ALL, "Rus");
	srand(time(NULL));
	float snow = 0;
	float rain = 0;
	const int n = 4;
	float r[n];
	float t[n];
	for (int i = 0; i < n; i++)
	{
		r[i] = rand() % 10;
		t[i] = rand() % 40 - 20;
		if (t[i] >= 0)
		{
			rain = rain + r[i];
		}
		if (t[i] < 0)
		{
		snow = snow + r[i];
		}
	}
	for (int i = 0; i < n; i++)
	{
		cout << "Temp " << t[i] << endl;
		cout << "Precip " << r[i] << endl;
		cout << "+" << endl;
	}
	
	cout << "Precipitation per month ("<<n<<" days)"<<endl;
	cout << "Snow: " << snow <<" mm." <<endl;
	cout << "Rain: " << rain <<" mm." << endl;
	system("pause");
	return 0;
}

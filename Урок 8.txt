#include <iostream>
#include <ctime>
#include <windows.h>
#include <iomanip>


using namespace std;

int main()
{


	setlocale(LC_ALL, "Russian");
	cout << "Enter" << endl;
	int n, c = 1;
	cin >> n;
	cout << "1, ";
	DWORD time1 = ::GetTickCount();
	for (int i = 1; i < n; i++)
	{
		c = c + i;
		cout << c << ", " <<  endl;
	}	
	DWORD time2 = ::GetTickCount() - time1;
	cout << setiosflags(ios::scientific) <<"Msec : "<< time2 << endl;
    system("pause");
}

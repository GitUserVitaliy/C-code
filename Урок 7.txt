#include <iostream>
#include <ctime>

using namespace std;
void main()
{
	setlocale(LC_ALL, "Rus");
	srand(time(NULL));

	float n = 0, sum = 0;
	while (true)
	{
		cin >> n;
		if (n < 0)
		{
			continue;
		}
		if (n == 0)
		{
			break;
		}
		sum = sum + n;
	}
	cout << sum << endl;

	system("pause");
}

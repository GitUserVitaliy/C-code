#include <iostream>
#include <ctime>

using namespace std;

int main()
{
	srand(time(NULL));
	const int x = 8,y = 10; //x - строки , y - столбцы; Будет смотрется интереснее если x = 28, а y = 59;
	int array[x][y];
	for (int i = 0; i < x; i++)
	{
		for (int j = 0; j < y; j++)
		{
			array[i][j] = rand() % 2;
		}
	}
	cout << "Matrix: " << endl;
	for (int i = 0; i < x; i++)
	{
		for (int j = 0; j < y; j++)
		{
			cout << array[i][j] << " ";
		}
		cout << endl;
	}
	system("pause");
	return 0;
}

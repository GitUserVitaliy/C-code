#include <iostream>
#include <fstream>
#include <string>
#include <vector>


using namespace std;

int main()
{

	int sum = 0;
	int count = 0;
	ifstream fin;
	fin.open("input.txt");
	while(!fin.eof())
	{
		fin >> var;
		sum = sum + var;

}
